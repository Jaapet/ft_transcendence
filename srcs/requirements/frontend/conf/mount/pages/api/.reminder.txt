NOTES:
- Data.error_message can vary (detail, username, etc)

try {
	const response = await fetch(request);
	if (!response) {
		throw new Error('Fetch failed');
	}

	const data = await response.json();
	if (!data) {
		throw new Error('Fetch failed');
	}
	if (!response.ok) {
		throw new Error(data.error_message || 'Fetch failed');
	}
	// do stuff
